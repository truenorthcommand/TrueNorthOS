export class ReferralService {
  static async createReferralCode(
    ownerId: string,
    ownerType: 'customer' | 'merchant',
    landingType: 'customer' | 'merchant',
  ) {
    const existing = await db
      .select()
      .from(referralCodes)
      .where(and(eq(referralCodes.ownerId, ownerId), eq(referralCodes.ownerType, ownerType)))
      .limit(1);

    if (existing.length > 0) return existing[0];

    let code = '';
    let attempts = 0;
    const maxAttempts = 10;

    while (attempts < maxAttempts) {
      code = nanoid(6).toUpperCase();
      const dup = await db
        .select()
        .from(referralCodes)
        .where(eq(referralCodes.code, code))
        .limit(1);
      if (dup.length === 0) break;
      attempts += 1;
    }

    if (!code) throw new Error('Failed to generate referral code');

    const [created] = await db
      .insert(referralCodes)
      .values({ ownerId, ownerType, landingType, code })
      .returning();
    return created;
  }

  static async getReferralCodeByCode(code: string) {
    const [row] = await db
      .select()
      .from(referralCodes)
      .where(eq(referralCodes.code, code.toUpperCase()))
      .limit(1);
    return row;
  }

  static async getUserReferralCode(ownerId: string, ownerType: 'customer' | 'merchant') {
    const [row] = await db
      .select()
      .from(referralCodes)
      .where(and(eq(referralCodes.ownerId, ownerId), eq(referralCodes.ownerType, ownerType)))
      .limit(1);
    return row;
  }

  static hashValue(value: string) {
    return crypto.createHash('sha256').update(value).digest('hex');
  }

  static async logReferralEvent(
    referralCodeId: string,
    ip: string,
    userAgent: string,
    geoCountry?: string,
  ) {
    await db.insert(referralEvents).values({
      referralCodeId,
      ipHash: this.hashValue(ip),
      userAgentHash: this.hashValue(userAgent),
      geoCountry,
    });
  }

  static async createConversion(referralCodeId: string, referredUserId: string) {
    const existing = await db
      .select()
      .from(referralConversions)
      .where(eq(referralConversions.referredUserId, referredUserId))
      .limit(1);
    if (existing.length > 0) return existing[0];

    const [row] = await db
      .insert(referralConversions)
      .values({ referralCodeId, referredUserId, status: 'pending' })
      .returning();
    return row;
  }
}

// SECTION 5 – FRAUD DETECTION
// =======================================

export class FraudDetection {
  static async checkSuspiciousActivity(
    referralCodeId: string,
    ip: string,
    userAgent: string,
  ): Promise<{ suspicious: boolean; reason?: string }> {
    const ipHash = ReferralService.hashValue(ip);
    const userAgentHash = ReferralService.hashValue(userAgent);
    const oneDayAgo = new Date();
    oneDayAgo.setDate(oneDayAgo.getDate() - 1);

    const recentEvents = await db
      .select()
      .from(referralEvents)
      .where(
        and(
          eq(referralEvents.referralCodeId, referralCodeId),
          eq(referralEvents.ipHash, ipHash),
          eq(referralEvents.userAgentHash, userAgentHash),
          gte(referralEvents.scannedAt, oneDayAgo),
        ),
      );

    if (recentEvents.length > 5) {
      return { suspicious: true, reason: 'Too many scans from same IP/UA in 24h' };
    }

    const ipScans = await db
      .select({ count: sql<number>`count(*)` })
      .from(referralEvents)
      .where(and(eq(referralEvents.ipHash, ipHash), gte(referralEvents.scannedAt, oneDayAgo)));

    if (Number(ipScans[0]?.count ?? 0) > 20) {
      return { suspicious: true, reason: 'IP rate limit exceeded' };
    }

    return { suspicious: false };
  }

  static async checkDuplicatePaymentMethod(userId: string, paymentMethodId: string) {
    const existing = await db
      .select()
      .from(users)
      .where(
        and(
          eq(users.stripePaymentMethodId, paymentMethodId),
          sql`${users.id} != ${userId}`,
        ),
      )
      .limit(1);
    return existing.length > 0;
  }

  static async disqualifyConversion(conversionId: string, reason: string) {
    await db
      .update(referralConversions)
      .set({ status: 'disqualified' })
      .where(eq(referralConversions.id, conversionId));
    console.log('Disqualified conversion', conversionId, reason);
  }
}

// SECTION 6 – DISCOUNT ENGINE
// =======================================

export interface DiscountBreakdown {
  reviewDiscount: number;
  referralDiscount: number;
  fixedMonthlyCredit: number;
  totalPercentDiscount: number;
  effectiveDiscount: number;
  monthlySavings: number;
  cappedReferrals: number;
  creditPerCappedReferral: number;
  totalPostCapCredit: number;
}

export class DiscountEngine {
  private static readonly REVIEW_MAX_PERCENT = 20;
  private static readonly REFERRAL_MAX_PERCENT = 50;
  private static readonly COMBINED_CAP_PERCENT = 50;
  private static readonly POST_CAP_CREDIT_PER_REFERRAL = 25;

  static async calculateUserDiscount(userId: string, monthlyPrice: number): Promise<DiscountBreakdown> {
    const reviewRows = await db
      .select()
      .from(reviewRewards)
      .where(and(eq(reviewRewards.userId, userId), eq(reviewRewards.active, true)));

    let reviewDiscountPercent = 0;
    let fixedMonthlyCredit = 0;

    for (const rw of reviewRows) {
      const val = Number(rw.value);
      if (rw.valueType === 'percent') reviewDiscountPercent += val;
      else fixedMonthlyCredit += val;
    }
    reviewDiscountPercent = Math.min(reviewDiscountPercent, this.REVIEW_MAX_PERCENT);

    const [userCode] = await db
      .select()
      .from(referralCodes)
      .where(and(eq(referralCodes.ownerId, userId), eq(referralCodes.ownerType, 'customer')))
      .limit(1);

    let qualifiedReferralCount = 0;
    if (userCode) {
      const [agg] = await db
        .select({ count: sql<number>`count(*)` })
        .from(referralConversions)
        .where(and(eq(referralConversions.referralCodeId, userCode.id), eq(referralConversions.status, 'qualified')));
      qualifiedReferralCount = Number(agg?.count ?? 0);
    }

    let referralDiscountPercent = Math.min(
      qualifiedReferralCount * 5,
      this.REFERRAL_MAX_PERCENT,
    );

    const totalPercentBeforeCap = reviewDiscountPercent + referralDiscountPercent;
    const effectivePercentDiscount = Math.min(
      totalPercentBeforeCap,
      this.COMBINED_CAP_PERCENT,
    );

    const referralsForCap = this.COMBINED_CAP_PERCENT / 5;
    const cappedReferrals = Math.max(0, qualifiedReferralCount - referralsForCap);

    const percentDiscountAmount = (monthlyPrice * effectivePercentDiscount) / 100;
    const totalMonthlyDiscount = percentDiscountAmount + fixedMonthlyCredit;
    const finalMonthlyDiscount = Math.min(totalMonthlyDiscount, monthlyPrice);
    const postCapCredit = cappedReferrals * this.POST_CAP_CREDIT_PER_REFERRAL;

    return {
      reviewDiscount: reviewDiscountPercent,
      referralDiscount: referralDiscountPercent,
      fixedMonthlyCredit,
      totalPercentDiscount: totalPercentBeforeCap,
      effectiveDiscount: effectivePercentDiscount,
      monthlySavings: finalMonthlyDiscount,
      cappedReferrals,
      creditPerCappedReferral: this.POST_CAP_CREDIT_PER_REFERRAL,
      totalPostCapCredit: postCapCredit,
    };
  }
}

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY ?? '', {
  apiVersion: '2024-11-20.acacia',
});

export class StripeService {
  static async applyMonthlyDiscount(userId: string, customerId: string, monthlyPrice: number) {
    const discount = await DiscountEngine.calculateUserDiscount(userId, monthlyPrice);
    if (discount.monthlySavings > 0) {
      await stripe.customers.update(customerId, {
        balance: Math.round(discount.monthlySavings * -100),
      });
    }
    if (discount.totalPostCapCredit > 0) {
      await stripe.customers.update(customerId, {
        metadata: {
          post_cap_credit: discount.totalPostCapCredit.toString(),
        },
      });
    }
  }

  static async createCreditInvoiceItem(customerId: string, amount: number, description: string) {
    return stripe.invoiceItems.create({
      customer: customerId,
      amount: Math.round(amount * -100),
      currency: 'gbp',
      description,
    });
  }
}

export async function processMonthlyCommissions() {
  const now = new Date();
  const periodMonth = `${now.getFullYear()}-${String(now.getMonth() + 1).padStart(2, '0')}`;
  console.log('Processing merchant commissions for', periodMonth);

  const threeMonthsAgo = new Date();
  threeMonthsAgo.setMonth(threeMonthsAgo.getMonth() - 3);

  const qualifiedMerchantConversions = await db
    .select({
      conv: referralConversions,
      code: referralCodes,
      user,
    })
    .from(referralConversions)
    .innerJoin(referralCodes, eq(referralConversions.referralCodeId, referralCodes.id))
    .innerJoin(users, eq(referralConversions.referredUserId, users.id))
    .where(
      and(
        eq(referralConversions.status, 'qualified'),
        eq(referralCodes.ownerType, 'merchant'),
        gte(referralConversions.qualifiedAt!, threeMonthsAgo),
      ),
    );

  for (const row of qualifiedMerchantConversions) {
    const { conv, code, user } = row as any;
    const existing = await db
      .select()
      .from(merchantEarnings)
      .where(
        and(
          eq(merchantEarnings.merchantId, code.ownerId),
          eq(merchantEarnings.sourceUserId, user.id),
          eq(merchantEarnings.type, 'signup_bonus'),
        ),
      )
      .limit(1);
    if (existing.length === 0) {
      await db.insert(merchantEarnings).values({
        merchantId: code.ownerId,
        sourceUserId: user.id,
        amount: '10.00',
        type: 'signup_bonus',
        periodMonth,
        paid: false,
      });
    }
  }

  const planPrices: Record<string, number> = {
    starter: 35,
    pro: 60,
    business: 150,
  };

  const recurring = await db
    .select({
      conv: referralConversions,
      code: referralCodes,
      user,
    })
    .from(referralConversions)
    .innerJoin(referralCodes, eq(referralConversions.referralCodeId, referralCodes.id))
    .innerJoin(users, eq(referralConversions.referredUserId, users.id))
    .where(
      and(
        eq(referralConversions.status, 'qualified'),
        eq(referralCodes.ownerType, 'merchant'),
        eq(users.subscriptionActive, true),
      ),
    );

  for (const row of recurring) {
    const { code, user } = row as any;
    const price = planPrices[user.plan] ?? 0;
    if (!price) continue;
    const commission = price * 0.05;

    const existing = await db
      .select()
      .from(merchantEarnings)
      .where(
        and(
          eq(merchantEarnings.merchantId, code.ownerId),
          eq(merchantEarnings.sourceUserId, user.id),
          eq(merchantEarnings.type, 'recurring_commission'),
          eq(merchantEarnings.periodMonth, periodMonth),
        ),
      )
      .limit(1);
    if (existing.length === 0) {
      await db.insert(merchantEarnings).values({
        merchantId: code.ownerId,
        sourceUserId: user.id,
        amount: commission.toFixed(2),
        type: 'recurring_commission',
        periodMonth,
        paid: false,
      });
    }
  }
}

interface ReferralStats {
  code: string;
  totalScans: number;
  totalSignups: number;
  qualifiedReferrals: number;
  currentDiscount: number;
  monthlySavings: number;
}

export function CustomerReferralDashboard() {
  const [qrDataUrl, setQrDataUrl] = useState('');
  const canvasRef = useRef<HTMLCanvasElement | null>(null);

  const { data: stats } = useQuery<ReferralStats>({
    queryKey: ['referral-stats'],
    queryFn: async () => {
      const res = await fetch('/api/referrals/stats');
      if (!res.ok) throw new Error('Failed to fetch referral stats');
      return res.json();
    },
  });

  const referralUrl = stats ? `${window.location.origin}/r/${stats.code}` : '';

  useEffect(() => {
    if (!referralUrl || !canvasRef.current) return;
    QRCode.toCanvas(canvasRef.current, referralUrl, { width: 200, margin: 2 });
    QRCode.toDataURL(referralUrl, { width: 400 }).then(setQrDataUrl);
  }, [referralUrl]);

  const copyLink = () => {
    if (!referralUrl) return;
    navigator.clipboard.writeText(referralUrl).catch(() => {});
  };

  const downloadQR = () => {
    if (!qrDataUrl) return;
    const a = document.createElement('a');
    a.href = qrDataUrl;
    a.download = 'truenorth-referral.png';
    a.click();
  };

  if (!stats) return <div>Loading...</div>;

  return (
    <div className="space-y-6">
      <h2 className="text-2xl font-bold">Referral Dashboard</h2>
      <div className="grid gap-4 md:grid-cols-4">
        <Card>
          <CardHeader>
            <CardTitle>Total Scans</CardTitle>
          </CardHeader>
          <CardContent>
            <div className="text-3xl font-bold">{stats.totalScans}</div>
          </CardContent>
        </Card>
        <Card>
          <CardHeader>
            <CardTitle>Signups</CardTitle>
          </CardHeader>
          <CardContent>
            <div className="text-3xl font-bold">{stats.totalSignups}</div>
          </CardContent>
        </Card>
        <Card>
          <CardHeader>
            <CardTitle>Qualified Referrals</CardTitle>
          </CardHeader>
          <CardContent>
            <div className="text-3xl font-bold">{stats.qualifiedReferrals}</div>
          </CardContent>
        </Card>
        <Card>
          <CardHeader>
            <CardTitle>Monthly Savings</CardTitle>
          </CardHeader>
          <CardContent>
            <div className="text-3xl font-bold text-green-600">
              £{stats.monthlySavings.toFixed(2)}
            </div>
            <p className="text-sm text-muted-foreground">{stats.currentDiscount}% discount</p>
          </CardContent>
        </Card>
      </div>

      <Card>
        <CardHeader>
          <CardTitle>Your Referral Link</CardTitle>
          <CardDescription>Share this link or QR code</CardDescription>
        </CardHeader>
        <CardContent>
          <div className="flex items-center gap-2 mb-4">
            <code className="flex-1 rounded bg-muted px-3 py-2 text-sm">{referralUrl}</code>
            <Button onClick={copyLink}>
              <Copy className="mr-1 h-4 w-4" /> Copy
            </Button>
          </div>

      </Card>
