Implement comprehensive audit logging system for TrueNorth OS:

1. CREATE FOUR DATABASE TABLES:

**audit_logs table (main audit log):**
```sql
CREATE TABLE audit_logs (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  
  -- When & Where
  timestamp DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  page_route VARCHAR(255),
  ip_address VARCHAR(45), -- IPv6 compatible
  user_agent TEXT,
  session_id VARCHAR(255),
  
  -- Who
  user_id INT NOT NULL,
  user_name VARCHAR(255) NOT NULL,
  user_email VARCHAR(255),
  user_role VARCHAR(100) NOT NULL, -- admin, dispatcher, manager, client, etc.
  
  -- What
  action_type VARCHAR(50) NOT NULL, -- create, update, delete, login, export, etc.
  action_category VARCHAR(50), -- auth, client, job, finance, settings, etc.
  entity_type VARCHAR(100) NOT NULL, -- client, job, quote, invoice, user, etc.
  entity_id VARCHAR(100), -- ID of affected record
  
  -- Details
  action_description TEXT, -- Human-readable description
  changes_json JSON, -- Full before/after data
  metadata_json JSON, -- Additional context
  
  -- Severity & Classification
  severity VARCHAR(20) DEFAULT 'info', -- info, warning, critical
  is_sensitive BOOLEAN DEFAULT FALSE,
  requires_review BOOLEAN DEFAULT FALSE,
  
  -- Audit trail integrity
  previous_log_id BIGINT,
  checksum VARCHAR(64), -- SHA-256 hash
  
  INDEX idx_timestamp (timestamp),
  INDEX idx_user_id (user_id),
  INDEX idx_action_type (action_type),
  INDEX idx_entity_type_id (entity_type, entity_id),
  INDEX idx_session (session_id),
  INDEX idx_severity (severity)
);
```

**user_sessions table:**
```sql
CREATE TABLE user_sessions (
  id INT PRIMARY KEY AUTO_INCREMENT,
  session_id VARCHAR(255) UNIQUE NOT NULL,
  user_id INT NOT NULL,
  login_timestamp DATETIME NOT NULL,
  logout_timestamp DATETIME,
  last_activity DATETIME,
  ip_address VARCHAR(45),
  device_info TEXT,
  is_active BOOLEAN DEFAULT TRUE,
  
  INDEX idx_user_id (user_id),
  INDEX idx_session_id (session_id),
  INDEX idx_active (is_active)
);
```

**failed_actions_log table:**
```sql
CREATE TABLE failed_actions_log (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  timestamp DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  user_id INT,
  attempted_email VARCHAR(255),
  action_attempted VARCHAR(255) NOT NULL,
  failure_reason TEXT,
  ip_address VARCHAR(45),
  user_agent TEXT,
  
  INDEX idx_timestamp (timestamp),
  INDEX idx_ip_address (ip_address),
  INDEX idx_attempted_email (attempted_email)
);
```

**audit_log_access table:**
```sql
CREATE TABLE audit_log_access (
  id INT PRIMARY KEY AUTO_INCREMENT,
  timestamp DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  accessed_by_user_id INT NOT NULL,
  accessed_by_user_name VARCHAR(255),
  access_type VARCHAR(50), -- view, export, search
  filters_applied JSON,
  records_accessed INT,
  export_format VARCHAR(20), -- csv, pdf, json
  
  INDEX idx_accessed_by (accessed_by_user_id),
  INDEX idx_timestamp (timestamp)
);
```
