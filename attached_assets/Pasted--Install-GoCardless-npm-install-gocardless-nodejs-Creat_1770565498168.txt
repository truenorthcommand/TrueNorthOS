cat > services/gocardless/redirectFlow.ts << 'EOF'
import gocardless from 'gocardless-nodejs';
import { db } from '@/db';
import { pendingMandates } from '@/db/schema';

const client = gocardless(
  process.env.GOCARDLESS_ACCESS_TOKEN!,
  gocardless.constants.Environments.Live
);

export const initiateSubscriptionSetup = async (
  userId: string,
  customerData: {
    email: string;
    firstName: string;
    lastName: string;
    addressLine1?: string;
    city?: string;
    postalCode?: string;
  }
) => {
  const sessionToken = `${userId}-${Date.now()}`;
  
  const redirectFlow = await client.redirectFlows.create({
    description: 'TrueNorth OS Maintenance Subscription',
    session_token: sessionToken,
    success_redirect_url: `${process.env.APP_URL}/billing/setup-complete`,
    prefilled_customer: {
      email: customerData.email,
      given_name: customerData.firstName,
      family_name: customerData.lastName,
      address_line1: customerData.addressLine1,
      city: customerData.city,
      postal_code: customerData.postalCode,
    },
  });
  
  await db.insert(pendingMandates).values({
    userId,
    sessionToken,
    redirectFlowId: redirectFlow.id,
    status: 'pending',
    expiresAt: new Date(Date.now() + 30 * 60 * 1000),
  });
  
  return {
    redirectUrl: redirectFlow.redirect_url,
    sessionToken,
  };
};
EOF

# Create completion service
cat > services/gocardless/completeSetup.ts << 'EOF'
import gocardless from 'gocardless-nodejs';
import { db } from '@/db';
import { pendingMandates, recurringPayments } from '@/db/schema';
import { and, eq, gt } from 'drizzle-orm';

const client = gocardless(
  process.env.GOCARDLESS_ACCESS_TOKEN!,
  gocardless.constants.Environments.Live
);

export const completeSubscriptionSetup = async (
  redirectFlowId: string,
  sessionToken: string,
  planDetails: {
    amount: number;
    frequency: 'monthly' | 'quarterly' | 'annually';
  }
) => {
  const pending = await db.query.pendingMandates.findFirst({
    where: and(
      eq(pendingMandates.sessionToken, sessionToken),
      eq(pendingMandates.redirectFlowId, redirectFlowId),
      eq(pendingMandates.status, 'pending'),
      gt(pendingMandates.expiresAt, new Date())
    ),
  });
  
  if (!pending) {
    throw new Error('Invalid or expired session');
  }
  
  const completedFlow = await client.redirectFlows.complete(
    redirectFlowId,
    { session_token: sessionToken }
  );
  
  const gcCustomerId = completedFlow.links.customer;
  const gcMandateId = completedFlow.links.mandate;
  
  const subscription = await client.subscriptions.create({
    amount: planDetails.amount,
    currency: 'GBP',
    name: `TrueNorth OS - ${planDetails.frequency}`,
    interval_unit: planDetails.frequency === 'annually' ? 'yearly' : 'monthly',
    interval: planDetails.frequency === 'quarterly' ? 3 : 1,
    metadata: {
      user_id: pending.userId,
    },
    links: {
      mandate: gcMandateId,
    },
  });
  
  const [payment] = await db.insert(recurringPayments).values({
    customerId: pending.userId,
    gcCustomerId,
    gcMandateId,
    gcSubscriptionId: subscription.id,
    amount: planDetails.amount,
    frequency: planDetails.frequency,
    status: 'active',
    nextPaymentDate: new Date(subscription.upcoming_payments[0].charge_date),
  }).returning();
  
  await db.update(pendingMandates)
    .set({ status: 'completed' })
    .where(eq(pendingMandates.id, pending.id));
  
  return {
    subscriptionId: subscription.id,
    nextPaymentDate: subscription.upcoming_payments[0].charge_date,
    payment,
  };
};
EOF

export const pendingMandates = pgTable('pending_mandates', {
  id: serial('id').primaryKey(),
  userId: text('user_id').notNull(),
  sessionToken: text('session_token').notNull().unique(),
  redirectFlowId: text('redirect_flow_id').notNull(),
  status: text('status').notNull().default('pending'),
  expiresAt: timestamp('expires_at').notNull(),
  createdAt: timestamp('created_at').defaultNow(),
});

export const recurringPayments = pgTable('recurring_payments', {
  id: serial('id').primaryKey(),
  customerId: text('customer_id').notNull(),
  gcCustomerId: text('gc_customer_id').notNull(),
  gcMandateId: text('gc_mandate_id').notNull(),
  gcSubscriptionId: text('gc_subscription_id'),
  amount: integer('amount').notNull(),
  frequency: text('frequency').notNull(),
  status: text('status').notNull().default('pending'),
  nextPaymentDate: timestamp('next_payment_date'),
  createdAt: timestamp('created_at').defaultNow(),
  updatedAt: timestamp('updated_at').defaultNow(),
});

import { NextRequest, NextResponse } from 'next/server';
import { initiateSubscriptionSetup } from '@/services/gocardless/redirectFlow';
import { auth } from '@/lib/auth';

export async function POST(req: NextRequest) {
  const session = await auth();
  if (!session?.user) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
  }
  
  const body = await req.json();
  
  try {
    const { redirectUrl, sessionToken } = await initiateSubscriptionSetup(
      session.user.id,
      {
        email: session.user.email!,
        firstName: body.firstName,
        lastName: body.lastName,
        addressLine1: body.addressLine1,
        city: body.city,
        postalCode: body.postalCode,
      }
    );
    
    return NextResponse.json({ redirectUrl, sessionToken });
  } catch (error) {
    console.error('GoCardless setup failed:', error);
    return NextResponse.json(
      { error: 'Failed to initiate setup' },
      { status: 500 }
    );
  }
}

cat > app/api/billing/complete/route.ts << 'EOF'
import { NextRequest, NextResponse } from 'next/server';
import { completeSubscriptionSetup } from '@/services/gocardless/completeSetup';

export async function POST(req: NextRequest) {
  const { searchParams } = new URL(req.url);
  const redirectFlowId = searchParams.get('redirect_flow_id');
  
  const body = await req.json();
  const { sessionToken, amount, frequency } = body;
  
  if (!redirectFlowId || !sessionToken) {
    return NextResponse.json({ error: 'Missing parameters' }, { status: 400 });
  }
  
  try {
    const result = await completeSubscriptionSetup(
      redirectFlowId,
      sessionToken,
      { amount, frequency }
    );
    
    return NextResponse.json(result);
  } catch (error) {
    console.error('GoCardless completion failed:', error);
    return NextResponse.json(
      { error: 'Failed to complete setup' },
      { status: 500 }
    );
  }
}

'use client';

import { Button } from '@/components/ui/button';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import { RadioGroup, RadioGroupItem } from '@/components/ui/radio-group';

export const SubscriptionSetup = () => {
  const [loading, setLoading] = useState(false);
  const [plan, setPlan] = useState<'monthly' | 'quarterly' | 'annually'>('monthly');
  const [customerData, setCustomerData] = useState({
    firstName: '',
    lastName: '',
    addressLine1: '',
    city: '',
    postalCode: '',
  });
  
  const prices = {
    monthly: 4900,
    quarterly: 13900,
    annually: 52900,
  };
  
  const handleSetup = async () => {
    setLoading(true);
    
    try {
      const response = await fetch('/api/billing/setup', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          ...customerData,
          amount: prices[plan],
          frequency: plan,
        }),
      });

        sessionToken,

              <RadioGroupItem value="monthly" id="monthly" />

'use client';

import { useEffect, useState } from 'react';
import { useSearchParams } from 'next/navigation';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { CheckCircle, XCircle } from 'lucide-react';

export default function SetupCompletePage() {
  const searchParams = useSearchParams();
  const [status, setStatus] = useState<'loading' | 'success' | 'error'>('loading');
  const [message, setMessage] = useState('');
  
  useEffect(() => {
    const completeSetup = async () => {
      const redirectFlowId = searchParams.get('redirect_flow_id');
      
      const sessionData = localStorage.getItem('gc_session');
      if (!sessionData || !redirectFlowId) {
        setStatus('error');
        setMessage('Invalid session. Please try again.');
        return;
      }
      
      const { sessionToken, amount, frequency } = JSON.parse(sessionData);
      
      try {
        const response = await fetch(
          `/api/billing/complete?redirect_flow_id=${redirectFlowId}`,
          {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ sessionToken, amount, frequency }),
          }
        );
        
        if (!response.ok) throw new Error('Setup failed');
        
        const result = await response.json();
        
        setStatus('success');
        setMessage(`Subscription active! Next payment: ${result.nextPaymentDate}`);
        
        localStorage.removeItem('gc_session');
        
        setTimeout(() => {
          window.location.href = '/dashboard';
        }, 3000);
        
      } catch (error) {
        setStatus('error');
        setMessage('Failed to complete setup. Please contact support.');
      }
    };
    
    completeSetup();
  }, [searchParams]);
  
  return (
    <div className="container max-w-md mx-auto py-20">
      <Card>
        <CardHeader>
          <CardTitle className="flex items-center gap-2">
            {status === 'loading' && 'Completing Setup...'}
            {status === 'success' && (
              <>
                <CheckCircle className="text-green-500" />
                Setup Complete
              </>
            )}
            {status === 'error' && (
              <>
                <XCircle className="text-red-500" />
                Setup Failed
              </>
            )}
          </CardTitle>
        </CardHeader>
        <CardContent>
          <p>{message}</p>
        </CardContent>
      </Card>
    </div>
  );
}
